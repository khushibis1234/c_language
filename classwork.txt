1)print 1 to 100

#include<stdio.h>
int main(){
    int i;
    for(i=1;i<=100;i++){
        printf("\n%d",i);
    }


return 0;
}

2) print 100 to 10(for loop)

#include<stdio.h>
int main(){
    int i;
    for(i=100;i>=10;i--){
        printf("\n%d",i);
    }


return 0;
